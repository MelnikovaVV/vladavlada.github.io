@startuml
actor Клиент
Клиент -> Tilda : Просматривает каталог товаров 
Клиент -> Tilda : Оформляет заказ
 
Tilda -> Сервер : Передает информацию о совершении заказа

entity Сервер

database БД 
Сервер -> БД : Передает информацию о заказанных товарах 
Сервер <- БД : Передает информацию о статусе подтверждения заказа 
Сервер -> Клиент : Подтверждение заказа 
 
actor Менеджер 
Менеджер -> БД : Проверяет заказы 
Менеджер -> БД : Обработка заказов
Менеджер <- БД : Получает информацию о заказах 
Клиент <-> Менеджер : Подтверждение заказа или обсуждение замены товаров 

@endum1
