@startuml
actor Клиент
actor Менеджер 
entity Почта


Клиент -> Tilda : Просматривает каталог товаров 
activate Tilda
Клиент <- Tilda : Каталог товаров на экране


 
Tilda -> CRM : Передает информацию \nо совершении заказа, \nуточняет наличие товара
activate CRM
Tilda <- CRM : Заказ принят в работу 
Клиент -> Tilda : Оформляет заказ 
Клиент <- Tilda : Заказ оформлен и \nпередан в обработку
Tilda -> Почта : Информация о новом заказе
Tilda <- Почта : Информация о заказе \nдоставлена до менеджера
deactivate Tilda
Менеджер -> Почта : Проверяет заказы 
Менеджер <- Почта : Получает информацию о заказах 


Менеджер -> CRM : Подтверждение наличия товара
CRM -> Менеджер : Заказ подтвержден
deactivate CRM

Менеджер -> Клиент : Обсуждение наличия товаров \nи способов оплаты и доставки
Менеджер <- Клиент : Согласование заказа, \nоплаты и доставки
 


@endum1
